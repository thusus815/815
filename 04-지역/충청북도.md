## 충북 출신 인물 (출생지·본적 기준)

```dataview
TABLE 출생지, 본적, side, 작위
FROM "01-인물"
WHERE contains(출생지, "충청북도") OR contains(출생지, "충북") 
   OR contains(본적, "충청북도") OR contains(본적, "충북")
   OR contains(출생지, "청주") OR contains(출생지, "충주") OR contains(출생지, "제천")
   OR contains(출생지, "보은") OR contains(출생지, "옥천") OR contains(출생지, "영동")
   OR contains(출생지, "진천") OR contains(출생지, "괴산") OR contains(출생지, "음성")
   OR contains(출생지, "단양") OR contains(출생지, "증평")
SORT side ASC, file.name ASC
LIMIT 200
```
